## 🔴 核心安全规则（必须严格遵守）

### 规则 1：命令串行 + 强制随机间隔

```
┌─────────────────────────────────────────────────────────────┐
│  ❌ 严禁并发执行任何 CLI 命令                                  │
│  ❌ 严禁使用固定间隔（必须每次重新随机）                         │
│  ❌ 严禁跳过 sleep（即使"感觉"不需要）                          │
│                                                              │
│  ✅ 所有命令必须逐条串行执行                                   │
│  ✅ 写操作每条命令完成后必须等待 40-55 秒随机间隔                │
│  ✅ 读操作每条命令完成后必须等待 30-50 秒随机间隔                │
│  ✅ 每次等待前必须重新生成随机数                                │
└─────────────────────────────────────────────────────────────┘
```

**强制执行模板**（复制使用，不要修改结构）：

```bash
# ═══ 执行单条命令的模板（写操作）═══
execute_with_delay() {
  local cmd="$1"
  local type="${2:-write}"  # write 或 read
  local delay
  
  if [ "$type" = "read" ]; then
    delay=$((30 + RANDOM % 21))  # 30-50 秒随机值
  else
    delay=$((40 + RANDOM % 16))  # 40-55 秒随机值
  fi
  
  echo "▶ 执行: $cmd"
  eval "$cmd"
  
  if [ $? -eq 0 ]; then
    echo "✓ 成功，等待 ${delay} 秒..."
    sleep "$delay"
  else
    echo "✗ 失败，跳过等待"
  fi
}
```

### 规则 2：不重复回复

```
┌─────────────────────────────────────────────────────────────┐
│  ❌ 同一条评论 cid 一生只能被回复一次（跨日跨轮均生效）           │
│  ❌ 同一作者短期内不重复（≥ 7 天冷却期）                        │
│                                                              │
│  ✅ 每次回复前必须检查 cid 是否在 REPLIED_CIDS 中              │
│  ✅ 回复成功后必须将 cid 加入 REPLIED_CIDS                    │
│  ✅ 必须检查同一作者的最后互动时间（≥ 7 天）                    │
└─────────────────────────────────────────────────────────────┘
```

**强制执行流程**：

```bash
# ═══ 检查是否已回复 ═══
is_replied() {
  local cid="$1"
  grep -q "$cid" /tmp/replied_cids.txt 2>/dev/null
  return $?
}

# ═══ 检查作者冷却期（≥ 7 天）═══
check_author_cooldown() {
  local user_id="$1"
  local last_interaction=$(node cli.js profile "$user_id" | grep -o '"last_seen":"[^"]*"' | cut -d'"' -f4)
  
  if [ -z "$last_interaction" ]; then
    return 0  # 无记录，可以互动
  fi
  
  local last_ts=$(date -d "$last_interaction" +%s 2>/dev/null || echo 0)
  local now_ts=$(date +%s)
  local days_diff=$(( (now_ts - last_ts) / 86400 ))
  
  if [ $days_diff -ge 7 ]; then
    return 0  # 冷却期已过
  else
    echo "⏭ 跳过：作者 $user_id 冷却期未过（${days_diff}/7 天）"
    return 1  # 冷却期未过
  fi
}

# ═══ 回复前检查 ═══
if is_replied "target_cid"; then
  echo "⏭ 跳过：已回复过该评论"
elif ! check_author_cooldown "target_user_id"; then
  echo "⏭ 跳过：作者冷却期未过"
else
  execute_with_delay 'node cli.js post <aweme_id> "内容" --reply-to target_cid'
  echo "target_cid" >> /tmp/replied_cids.txt  # 记录已回复
fi
```
